interface IApeMessage {
  type: 'user' | 'assistant' | 'system';
  content: string;
  timestamp: number;
  id?: string;
}

// 메시지 추가 함수
function addMessage(type: string, content: string) {
  const message = { type, content, timestamp: Date.now() };
  
  // 상태에 추가
  messageState.messages.push(message);
  vscode.setState(messageState);
  
  return message;
}
